const axios = require('axios');

const apiUrl = 'https://votredomaine.com/api/v1/index.php';
const apiKey = 'VOTRE_CLE_API';

const data = new URLSearchParams();
data.append('api_key', apiKey);
data.append('action', 'push');
data.append('service', 10);
data.append('link', 'https://instagram.com/p/XYZ123');
data.append('quantity', 1000);

axios.post(apiUrl, data)
  .then(response => {
    console.log(response.data);
  })
  .catch(error => {
    console.error(error);
  });
